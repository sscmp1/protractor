 *
 * @created    [04.08.2015] Lilia Sapurina.

describe('ps-date-time-picker_scroll_time', function () {

  var config = browser.params;

  // Глобальные переменные
  var button_xpath = config.button_xpath;
  var scrolling_element_xpath = config.scrolling_element_xpath;
  var time_under_scroll_xpath = config.time_under_scroll_xpath;
  var move_direction = config.move_direction;
  var move_size = config.move_size;

  // Поиск по локатору
  var date_time_picker_button = element(by.xpath(button_xpath));
  var scrolling_element = element(by.xpath(scrolling_element_xpath));
  var time_under_scroll = element(by.xpath(time_under_scroll_xpath));

  beforeEach(function(){
    browser.get('ng-components/examples/ps-date-time-picker.html');
  });

  it('при передвижении скроллера время остаётся в интервале от 00:00 до 23:59', function () {

    date_time_picker_button.click();
    // Перемещаем scroller в зависимости от параметра move_direction вправо или влево
    if (scrolling_element.isPresent()) {
      browser.actions().
              dragAndDrop(scrolling_element, {x: move_size * move_direction, y: 0}).
              perform();
      var max_time = '23:59';
      var min_time = '00:00';
      // Проверим, что время <= 23:59 и >=00:00
      time_under_scroll.getText().then(function (value) {
        expect(value <= max_time && value >= min_time).toBe(true);
      });
    }
    else {
      expect(true).toBe(false);
    }

  });

});

 describe('ps-date-time-picker_click_cross', function() {
   // Используем параметры, описанные в конфигурационном файле
   var config = browser.params;

   // Глобальные переменные
   var button_xpath = config.button_xpath;
   var calendar_xpath = config.calendar_xpath;
   var calendar_cross_xpath = config.calendar_cross_xpath;

   // Поиск по локатору
   var date_time_picker_button = element(by.xpath(button_xpath));
   var calendar = element(by.xpath(calendar_xpath));
   var calendar_cross = element(by.xpath(calendar_cross_xpath));

   beforeEach(function(){
     browser.get('ng-components/examples/ps-date-time-picker.html');
   });

   it('при нажатии на крестик календарь закрывается',function(){
     date_time_picker_button.click();
     calendar_cross.click();
     expect(calendar.isDisplayed()).toBe(false);
   });

 });

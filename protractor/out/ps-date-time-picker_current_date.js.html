describe('ps-date-time-picker_current_date', function() {

  var config = browser.params;

  // Глобальные переменные
  var calendar_field_xpath = config.calendar_field_xpath;

  // Поиск по локатору
  var calendar_field = element(by.xpath(calendar_field_xpath));

  beforeEach(function(){
    browser.get('ng-components/examples/ps-date-time-picker.html');
  });

  it('при открытии страницы в поле находится текущая дата и время',function(){

    calendar_field.getAttribute("value").then(function (field_value) {
      var today = new Date();
      // Приведём к нужному формату
      var dd = today.getDate();
      // Январь - месяц №0
      var mm = today.getMonth()+1;
      var yyyy = today.getFullYear();
      var hh = today.getHours();
      var min = today.getMinutes();

      if(dd<10){
        dd='0'+dd;
      }

      if(min<10){
        min='0'+min;
      }

      if(mm<10){
        mm='0'+mm;
      }

      if(hh<10){
        hh='0'+hh;
      }

      today = dd+'.'+mm+'.'+yyyy+' '+hh+':'+min;
      expect(field_value).toEqual(today);
    });

  });

});

describe('ps-grid_drop_down', function() {

  var config = browser.params;

  // Глобальные переменные
  var column_number = config.column_number;
  var columns_xpath = config.columns_xpath;
  var drop_down_xpath = config.drop_down_xpath;

  // Поиск по локатору
  var all_columns = element.all(by.xpath(columns_xpath));
  var drop_down = element(by.xpath(drop_down_xpath));

  var current_column = all_columns.get(column_number - 1);

  beforeEach(function(){
    browser.get('sbms/shell.html?shell_login=CMS_HAS&shell_password=qwerty&shell_modus=t&shell_modus=c');
    browser.executeScript("icms.go('WEB_INQ_PROC', 'InquiryList', null, 0)");
  });

  it('после клика правой кнопкой мыши появляется список', function() {
    // Кликнем правой кнопкой мыши
    browser.actions().mouseMove(current_column).click(protractor.Button.RIGHT).perform();

    // Проверим, что элемент отображается
    if (drop_down.isPresent()) {
      expect(drop_down.isDisplayed()).toBe(true);
    }

  });

});

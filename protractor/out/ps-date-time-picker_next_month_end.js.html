describe('ps-date-time-picker_next_month_end', function () {

  var config = browser.params;

  // Глобальные переменные
  var button_xpath = config.button_xpath;
  var calendar_field_xpath = config.calendar_field_xpath;
  var calendar_dates_repeater = config.calendar_dates_repeater;

  // Поиск по локатору
  var date_time_picker_button = element(by.xpath(button_xpath));
  var calendar_field = element(by.xpath(calendar_field_xpath));
  var calendar_dates = element.all(by.repeater(calendar_dates_repeater));

  beforeEach(function () {
    browser.get('ng-components/examples/ps-date-time-picker.html');
  });

  it('проверка всех полей из выпадающего списка: конец следующего месяца', function () {

    calendar_field.clear();
    calendar_field.sendKeys('Конец следующего месяца');
    date_time_picker_button.click();

    // Найдём на календаре отмеченный день и кликнем на него
    calendar_dates.filter(function (elem) {
      return elem.getAttribute("class").then(function (elem_class) {
        // Возвращаем дату с подстрокой в классе = selected
        return (elem_class.indexOf('selected') + 1);
      });
    }).then(function (filteredElements) {
      browser.actions().click(filteredElements[0]).perform();
    });

    calendar_field.getAttribute("value").then(function (field_value) {

      var today = new Date();

      // Приведём к нужному формату
      var mm = today.getMonth() + 2;
      var yyyy = today.getFullYear();

      // Количество дней в каждом месяце
      var day_quantity = [31,29,31,30,31,30,31,31,30,31,30,31];
      var dd;

      // Определяем последний день месяца (учитывая високостный год)
      if((mm === 2) && ((yyyy % 4)===0)){
        dd = day_quantity[mm - 1] - 1;
      }
      else{
        dd = day_quantity[mm - 1];
      }

      var hh = '23';
      var min = '59';

      if (mm < 10) {
        mm = '0' + mm;
      }

      today = dd + '.' + mm + '.' + yyyy + ' ' + hh + ':' + min;
      expect(field_value).toEqual(today);
    });

  });

});

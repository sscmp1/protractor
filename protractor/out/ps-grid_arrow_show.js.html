describe('ps-grid_arrow_show', function () {

  var config = browser.params;

  // Глобальные переменные
  var arrow_xpath = config.arrow_xpath;

  // Поиск по локатору
  var arrow = element(by.xpath(arrow_xpath));

  beforeEach(function(){
    browser.get('sbms/shell.html?shell_login=CMS_HAS&shell_password=qwerty&shell_modus=t&shell_modus=c');
    browser.executeScript("icms.go('WEB_INQ_PROC', 'InquiryList', null, 0)");
  });

  it('стрелка показывается', function () {

    if (arrow.isPresent()) {
      expect(arrow.isDisplayed()).toBe(true);
    }
    else {
      expect(true).toBe(false);
    }

  });

});

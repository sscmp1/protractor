describe('ps-date-time-picker_clear', function () {

  var config = browser.params;

  // Глобальные переменные
  var calendar_field_xpath = config.calendar_field_xpath;
  var clear_button_css = config.clear_button_css;

  // Поиск по локатору
  var calendar_field = element(by.xpath(calendar_field_xpath));
  var clear_button = element(by.css(clear_button_css));

  beforeEach(function(){
    browser.get('ng-components/examples/ps-date-time-picker.html');
  });

  it('при нажатии на кнопку с корзиной содержимое поля календаря очищается', function () {

    clear_button.click();
    expect(calendar_field.getAttribute("value")).toEqual('');

  });

});

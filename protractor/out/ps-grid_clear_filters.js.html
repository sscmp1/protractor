 *

 */

describe('ps-grid_clear_filters', function() {

  var config = browser.params;

  // Глобальные переменные
  var column_number = config.column_number;
  var columns_xpath = config.columns_xpath;
  var columns_filters_xpath = config.columns_filters_xpath;

  // Поиск по локатору
  var all_columns = element.all(by.xpath(columns_xpath));
  var columns_filters = element.all(by.xpath(columns_filters_xpath));

  var current_column = all_columns.get(column_number - 1);

  beforeEach(function(){
    browser.get('sbms/shell.html?shell_login=CMS_HAS&shell_password=qwerty&shell_modus=t&shell_modus=c');
    browser.executeScript("icms.go('WEB_INQ_PROC', 'InquiryList', null, 0)");
  });

  it('после клика на пункт "очистить все фильтры" все фильтры должны быть очищены', function() {

    // В первый фильтр (н.у.о.) отправим какое-нибудь значение
    columns_filters.get(0).sendKeys("Hello!");
    browser.sleep(3000);

    // Кликнем правой кнопкой мыши (путём эмулирования нажатия клавиши вниз дойдём до нужного пункта и нажмём enter)
    browser.actions().mouseMove(current_column).click(protractor.Button.RIGHT).sendKeys(protractor.Key.DOWN).sendKeys(protractor.Key.DOWN).sendKeys(protractor.Key.DOWN).sendKeys(protractor.Key.ENTER).perform();

    columns_filters.each(function (elm) {
      expect(elm.getAttribute("value")).toEqual("");
    });

  });

});
